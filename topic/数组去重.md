方法一：双层循环

```js
function unique(arr) {
    const res = [];
    for (let i = 0; i < arr.length; i ++) {
        if (res.indexOf(arr[i]) === -1) {
            res.push(arr[i]);
        }
    }
    return arr;
}
```

方法二：排序后去重

```js
function unique(arr) {
    const res = [];
    const tempArr = arr.concat().sort();
    let previous = null;

    for (let i = 0; i < tempArr.length; i ++>) {
        if (!i || previous !== tempArr[i]) {
            res.push(tempArr[i]);
        }
        previous = tempArr[i];
    }

    return res;
}
```
